    ## first level completion => task sub-command completion (( $+functions )) || _task_commands() {

    local cmd ret=1
    if (( CURRENT == 1 )); then
            # update IDs
            _task_zshids=( ${(f)"$(task _zshids)"} )

            _describe -t commands 'task command' _task_zshcmds
            _describe -t values 'task IDs' _task_zshids
            # TODO match more than one ID
    elif [[ $words[1] =~ ^[0-9]*$ ]] then
             _call_function ret _task_modify
             return ret
    else

    # local curcontext=“${curcontext}” # cmd=“${_task_cmds[®$words:*]%%:*}”

              cmd="${_task_cmds[(r)$words[1]]}"
          idCmd="${(M)_task_idCmds[@]:#$words[1]}"
if (( $#cmd )); then

    # curcontext=“${curcontext%:*:*}:task-${cmd}”

                                if (( $#idCmd )); then
                                    _call_function ret _task_id
                            else
                                    _call_function ret _task_${cmd} ||
                                            _call_function ret _task_default ||
                                                    _message "No command remaining."
                            fi
    else
        _message "Unknown subcommand ${cmd}"
    fi
    return ret
fi
